#### [e326 v2.py](<../my-submissions/e326 v2.py>)
```Python
class Solution:
    def isPowerOfThree(self, n: int) -> bool:
        if n <= 0 :
            return False
        while n != 1 :
            prev = n
            n //= 3
            if n * 3 != prev :
                return False

        return True
```

class Solution:
    def isPowerOfThree(self, n: int) -> bool:
        return n >= 1 and (n == 1 or n // 3 * 3 == n and self.isPowerOfThree(n // 3))
```

class Solution:
    def isPowerOfThree(self, n: int) -> bool:
        return n > 0 and n == 3 ** int(log(n, 3) + 0.0001)
```
