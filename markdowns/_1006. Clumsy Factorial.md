```C
int clumsy(int n) {
    switch (n) {
        case 4 :
            return 7;
        case 3 :
            return 6;
        case 2 :
        case 1 :
        case 0 :
            return n;
        default :
            break;
    }

    switch (n % 4) {
        case 3 :
            return n - 1;
        case 2 :
        case 1 :
            return n + 2;
        default :
            return n + 1;
    }
}
```

```Python
class Solution:
    def clumsy(self, n: int) -> int:
        match n :
            case 4 :
                return 7
            case 3 :
                return 6
            case 2 | 1 | 0 :
                return n
            
        output = n + 1

        match n % 4 :
            case 3 :
                output -= 2
            case 2 :
                output += 1
            case 1 :
                output += 1
            case 0 : 
                output = output
        return output
```
