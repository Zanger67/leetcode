```Python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def hasCycle(self, head: Optional[ListNode]) -> bool:
        visited = set()

        if not head :
            return False

        currNode = head
        while currNode.next :
            visited.add(currNode)
            currNode = currNode.next
            if currNode in visited :
                return True

        return False
```

### C
#### [e141 two pointer race.c](<../my-submissions/e141 two pointer race.c>)
```C
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
bool hasCycle(struct ListNode *head) {
    // Since no dynamic space and I don't feel like implementing a full datastructure,
    // let's do a 2-pointer method

    if (!head || !head->next) {
        return false;
    }

    struct ListNode *one = head;
    struct ListNode *two = head->next;

    while (one != two) {
        if (!two->next || !two->next->next) {
            return false;
        }
        one = one->next;
        two = two->next->next;
    }

    return true;
    
}
```
