class Solution:
    def findMatrix(self, nums: List[int]) -> List[List[int]]:
        cnt = Counter(nums)
        return [[x for x in cnt if cnt.get(x) >= c] for c in range(1, max(cnt.values()) + 1)]

```

class Solution:
    def findMatrix(self, nums: List[int]) -> List[List[int]]:
        cnt = Counter(nums)
        maxx = max(cnt.values())
        output = []
        for i in range(maxx) :
            output.append([])
        for i in cnt :
            for j in range(cnt.get(i)) :
                output[j].append(i)

        return output

```
