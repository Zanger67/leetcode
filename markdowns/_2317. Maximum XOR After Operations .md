### C
#### [m2317.c](<../my-submissions/m2317.c>)
```C
int maximumXOR(int* nums, int numsSize) {
    int output = 0;
    for (int i = 0; i < numsSize; i++) {
        output |= nums[i];
    }
    return output;
}
```

### C++

```C++
class Solution {
public:
    int maximumXOR(vector<int>& nums) {
        int output = 0;

        for (int num : nums) {
            output |= num;
        }

        return output;
    }
};
```
