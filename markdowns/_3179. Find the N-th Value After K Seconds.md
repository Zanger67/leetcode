class Solution:
    def valueAfterKSeconds(self, n: int, k: int) -> int:
        output = [1] * n
        
        for _ in range(k) :
            for j in range(1, n) :
                output[j] += output[j - 1]
        return output[-1] % (10 ** 9 + 7)
