class TwoSum:

    def __init__(self):
        self.vals = Counter()

    def add(self, number: int) -> None:
        self.vals[number] += 1

    def find(self, value: int) -> bool:
        return any(value - val in self.vals 
                   and (val != value - val or self.vals[value - val] > 1) for val in self.vals)
