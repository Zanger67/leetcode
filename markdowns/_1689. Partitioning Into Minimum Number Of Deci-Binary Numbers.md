#### [m1689.c](<../my-submissions/m1689.c>)
```C
int minPartitions(char* n) {
    int output = 0;
    
    while (*n) {
        output = *n - '0' < output ? output : (*n - '0');
        n++;
    }

    return output;
}
```

```Java
class Solution {
    public int minPartitions(String n) {
        int maxx = 0;

        for (char c : n.toCharArray()) {
            maxx = Integer.max(maxx, c - '0');
        }
        return maxx;
    }
}
```
