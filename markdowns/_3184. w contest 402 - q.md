```Python
class Solution:
    def countCompleteDayPairs(self, hours: List[int]) -> int:
        counter = 0
        
        for i in range(0, len(hours) - 1) :
            for j in range(i + 1, len(hours)) :
                if (hours[i] + hours[j]) % 24 == 0 :
                    counter += 1
                    
        return counter
```
