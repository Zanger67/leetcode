class Solution:
    def checkPowersOfThree(self, n: int) -> bool:
        return all((n % 3 ** (x + 1)) // (3 ** x) != 2 for x in range(int(n ** (1 / 3)), -1, -1))
```

class Solution:
    def checkPowersOfThree(self, n: int) -> bool:
        x = 1
        while x * 3 <= n :
            x *= 3

        while n :
            if n - 2 * x >= 0 :
                return False
            n %= x
            x //= 3

        return True
```
