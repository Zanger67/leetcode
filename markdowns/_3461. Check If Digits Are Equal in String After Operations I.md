class Solution:
    def hasSameDigits(self, s: str) -> bool:
        while len(s) > 2 :
            s_n = []
            for x, y in zip(s[:-1], s[1:]) :
                s_n.append(str((int(x) + int(y)) % 10))
            s = ''.join(s_n)
        
        return s[0] == s[1]
```

class Solution:
    def hasSameDigits(self, s: str) -> bool:
        a, b = 0, 0
        x = len(s) - 2
        for i, num in enumerate(s[:-1]) :
            a += (int(num) * comb(x, i)) % 100
        for i, num in enumerate(s[1:]) :
            b += (int(num) * comb(x, i)) % 100
        return a % 10 == b % 10
```
