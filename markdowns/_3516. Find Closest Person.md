#### [e3516.go](<../my-submissions/e3516.go>)
```Go
func findClosest(x int, y int, z int) int {
    a, b := x - z, y - z
    if a < 0 {
        a *= -1
    }
    if b < 0 {
        b *= -1
    }

    if a == b {
        return 0
    }
    if a < b {
        return 1
    }
    return 2
}
```

### Python
#### [e3516.py](<../my-submissions/e3516.py>)
```Python
class Solution:
    def findClosest(self, x: int, y: int, z: int) -> int:
        return 0 if (a := abs(x - z)) == (b := abs(y - z)) else 1 if a < b else 2
```
