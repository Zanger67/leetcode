```JavaScript
/**
 * @param {Function[]} functions
 * @return {Function}
 */
var compose = function(functions) {
    
    return function(x) {
        output = x
        for (let i = functions.length - 1; i >= 0 ; i--) {
            x = functions[i](x);
        }
        return x;
    }
};

/**
 * const fn = compose([x => x + 1, x => 2 * x])
 * fn(4) // 9
 */
```

var compose = function(functions) {
    
    return function(x) {
        output = x
        functions.reverse().forEach(el => output = el(output));
        return output;
    }
};
